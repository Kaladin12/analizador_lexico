X ::= ''

ESTRUCT ::= COND
ESTRUCT ::= LOOP
ESTRUCT ::= INTER

COND ::= COND_IF IF_CONTENT TEMP
TEMP ::= ''
TEMP ::= COND_ELSE CONTENT
IF_CONTENT ::= AGR_OP EXPR AGR_CP CONTENT
CONTENT ::= AGR_OB MAIN AGR_CB 
INTER ::= INTERRUPTOR AGR_OP EXPR AGR_CP AGR_OB CASE_ AGR_CB
CASE_ ::= CASO AGR_OP VALOR AGR_CP  AGR_OB MAIN ROMPER AGR_CB CASE_END
CASE_END ::= CASE_ 
CASE_END ::= DEFAULT 
DEFAULT ::= PREDETERMINADO AGR_OB MAIN ROMPER AGR_CB
LOOP ::= LOOP_FOR_ABS 
LOOP ::= LOOP_WHILE 
LOOP ::= LOOP_DO 
LOOP_FOR_ABS ::= CADA AGR_OP FORS
FORS ::= LOOP_PY ID
FORS ::= LOOP_FOR
LOOP_FOR ::= ASIGN EOL EXPR EOL EXPR AGR_CP  CONTENT
LOOP_WHILE ::= MNTRS  AGR_OP EXPR CONTENT EOL
LOOP_DO ::= HAZ_MNTRS CONTENT EXPR EOL
LOOP_PY ::= EN RANGO AGR_OP EXPR AGR_CP AGR_CP CONTENT
ASIGN ::= TIPO ID OP_ASIG EXPR EOL
